02

Output

おみくじルーレット

明るい状態から指でセンサーを覆い暗くすると回転開始。
指を離して明るくすると、徐々に減速して停止する仕組み。

Arduino — プログラム

const int cdsPin = A0;
const int stepPin = 9;
const int dirPin = 8;

const int threshold = 760;

void setup() {
  pinMode(stepPin, OUTPUT);
  pinMode(dirPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  int lightValue = analogRead(cdsPin);

  if (lightValue < threshold) {

    while (analogRead(cdsPin) < threshold) {
      stepMotor(1000);
    }

    for (int speed = 1000; speed < 8000; speed += 10) {
      stepMotor(speed);
    }
  }

  delay(100);
}

void stepMotor(int delayTime) {
  digitalWrite(dirPin, HIGH);
  digitalWrite(stepPin, HIGH);
  delayMicroseconds(delayTime);
  digitalWrite(stepPin, LOW);
  delayMicroseconds(delayTime);
}
